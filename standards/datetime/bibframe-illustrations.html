Example 1

This example illustrates a Contribution (including agent and role), identifiers Lccn and Isbn, hasInstance/instanceOf, and Text/Print as Work/Instance resource types.

 

 

<http://bibframe.example.org/work/4748136>

       a                                  bf:Work, bf:Text ;

 

       bf:title [                     a                              bf:Title ;

                                         rdfs:label               “One Hundred Armenian Tales” ] ;

 

      bf:contribution[      a                              bf:Contribution ;

                                        bf:agent                 <http://example.org/4748136#Agent700-23> ] ;

                                        bf:role                    <http://id.loc.gov/vocabulary/relators/ctb> ] ;

 

       bf:identifiedBy[     a                               bf:Lccn ;

                                       rdf:value                  “84673072” ] ;

 

      bf:hasInstance      <http://bibframe.example.org/instance/4748136> .

 

 

<http://example.org/4748136#Agent700-23 >

      a                             bf:Agent , bf:Person ;

      rdfs:label              "Villa, Susie Hoogasian" .

 

<http://id.loc.gov/vocabulary/relators/ctb>

      a                           bf:Role ;

     rdfs:label             "contributor" .

 

 

<http://bibframe.example.org/instance/4748136>

      a                        bf:Instance , bf:Print

      bf:instanceOf     < http://bibframe.example.org/work/4748136> ;

 

      bf:title [                     a                            bf:Title ;

                                        rdfs:label              “One Hundred Armenian Tales” ] ;

 

    bf:identifiedBy  [    a                              bf:Identifier , bf:Isbn ;

                                      rdf:value               “0814312829” ] ] .
